사용 예시

// 기본 사용법
const manager = new AoTMultiSourceManager(map, {
    apiKeys: { maptiler: 'YOUR_KEY' },
    enableMultiLayer: true
});

// 소스 전환
manager.switchSource('maptiler_vector');

// 오버레이 추가
manager.addSourceOverlay('vworld_wmts');

// 이벤트 리스닝
manager.on('sourceChange', (data) => {
    console.log('Active sources:', data.activeSources);
});
